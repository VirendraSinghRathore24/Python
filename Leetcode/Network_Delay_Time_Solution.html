    public int NetworkDelayTime(int[][] times, int n, int k)
    {
        Dictionary> adj = new Dictionary>();

        for(int i = 1; i <= n; i++)
        {
            adj[i] = new List();
        }

        foreach(var e in times)
        {
            int u = e[0];
            int v = e[1];
            int t = e[2];

            var list = adj[u];
            list.Add(new Pair(v, t));

            adj[u] = list;
        }

        Queue queue = new Queue();
        queue.Enqueue(new Pair(k, times[k][2]));

        List visited = new List();
        visited.Add(k);

        int min = -1;
        int count = 0;
        while(queue.Count > 0)
        {
            int size = queue.Count;

            count += size;
            for(int i = 0; i < size; i++)
            {
                var q = queue.Dequeue();
                min = q.t;

                foreach(var e in adj[q.v])
                {
                    if(!visited.Contains(e.v))
                    {
                        queue.Enqueue(new Pair(e.v, q.t + e.t));
                        visited.Add(e.v);
                    }
                }
            }
        }

        return count == n ? min : -1;
    }
      public class Pair
      {
        public int v;
        public int t;

        public Pair(int v, int t)
        {
            this.v = v;
            this.t = t;
        }
      }
